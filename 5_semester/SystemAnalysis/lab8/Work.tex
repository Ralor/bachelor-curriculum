	\chapter{Комбинаторные методы композиции}
	\section{Описание}
		\paragraph{Цель работы}
			Получить практические навыки в формировании вариа н-тов систем методом 
			морфологического анализа, в генерировании функций системы методом 
			структурно-функционального проектирования Казарновского, в 
			формировании множества функций управления системой.
		\paragraph{Формируемые компетенции}
			\begin{itemize}
			\item Владение культурой мышления, способность к обобщению, анализу, 
			восприятию информации, постановке цели и выбору путей е? достижения (ОК-1).
			\item Способность находить организационно-управленческие решения в не-стандартных
			ситуациях и готовность нести за них ответственность (ОК-4).
			\end{itemize}		
		\paragraph{Самостоятельная работа}
			Изучение метода морфологического анализа, метода структурно-функционального
			проектирования Казарновского, пере-борного метода формирования функций управления.

		\paragraph{Генерирование функций системы методом Казарновского}
			\subparagraph{Выбор объекта для структурно-функционального проектирования}
				Студия разработки компьютерных игр.
			\subparagraph{Список основных функций} 
			\begin{itemize}
			\item[] \(h\) - выпуск продукции, поддержка прод. лицензий
			\item[] \(v\) - поддержание элементов системы
			\item[] \(p\) - организация (адаптация к внешн. воздействию)
			\item[] \(c\) - управление
			\item[] \(f\) - обновление
			\end{itemize}
			\subparagraph{Интерпретации комбинированных функций}
			\begin{itemize}
			\item[] \(vh\) - сопровождение проданных лицензий, продажа новых
			\item[] \(ph\) - выход в новые сети дистрюьюции, реклама
			\item[] \(ch\) - использование методологий разработки (rup, agile)
			\item[] \(fh\) - выход на новые платформы, R \& D
			\item[] \(pf\) - участие во встречах разработчиков, изучение публикаций др. компаний
			\item[] \(cvh\) - использование CRM-систем
			\item[] \(vpf\) - публикация своих методик, интересных результатов, 
			технологических обобщений
			\item[] \(vfh\) - покупка сторонних разработок, новых устройств
			\item[] \(fpch\) - создание новых методологий разработки и их применение
			\item[] \(pfvh\) - помощь маркетологам в соц. исследованиях
			\end{itemize}
			\subparagraph{Подфункции по обеспечению предметами деятельности}
			\begin{itemize}
			\item[] \(i\) - обеспечение предметами деятельности
			\item[] \(k\) - обеспечение инструентами
			\item[] \(l\) - обеспечение энергией
			\item[] \(o\) - вывод продукции
			\item[] \(t\) - технологические преобразования
			\end{itemize}
			\subparagraph{Интерпретации комбинированных функций}
			\begin{itemize}
			\item[] \(tch\) - переход на новую методологию разработки
			\item[] \(iph\) - заключ. договоров с издателями, дистрибьюторами, рекламщиками
			\item[] \(vtf\) - приобретение ПО для моделирования физич. процессов и других сложн. расчетов
			\item[] \(kvpf\) - использование услуг сторонни специалистов
			\item[] \(ctvpf\) - управление видами/направленностью публикаций
			\end{itemize}
		\paragraph{Формирование множества функций управления системой}
			\subparagraph{Выбор объекта (производственной системы)}
			\begin{itemize}
			\item[] \(p_1\) - выбор целевой аудитории
			\item[] \(p_2\) - выбор платформы/игрового движка
			\item[] \(p_3\) - выбор жанра, стилистики
			\item[] \(p_4\) - проектирование/создание договоренностей/разработка
			\item[] \(p_5\) - публикация в магазине/на сайте/на дисках
			\item[] \(p_6\) - поддержка/выпуск дополнений
			\end{itemize}
			\subparagraph{Формирование исходных множеств}
			\begin{itemize}
			\item[] \(z_1\) - прогнозирование
			\item[] \(z_2\) - планирование
			\item[] \(z_3\) - организация
			\item[] \(z_4\) - учет
			\item[] \(z_5\) - контроль
			\end{itemize}
			\begin{itemize}
			\item[] \(x_1\) - изучение рынка, рез-тов чужих иссл., принятых стандартов
			\item[] \(x_2\) - поиск схожих проектов, анализ их опыта/результатов
			\item[] \(x_3\) - расчет возможностей, рисков, выбор лучших результатов
			\item[] \(x_4\) - анализ отзывов, поиск решений проблем
			\end{itemize}
			\subparagraph{Формирование функций управления}
			\begin{itemize}
			\item[] \(p_1 z_1\) - выбор платежеспособных игроков
			\item[] \(p_2 z_1\) - выбор подходящей для команды платформы
			\item[] \(p_3 z_2\) - обозначение темы, концепции, постановка сложных задач
			\item[] \(p_4 z_2\) - календарь разработки, первые дедлайны
			\item[] \(p_4 z_3\) - регулярная работа
			\item[] \(p_5 z_4\) - проверка предположений о рынке
			\item[] \(p_6 z_5\) - исправление багов, наполнение игр новым контентом
			\item[] \(p_1 z_1 x_1\) - выбор категории игроков по возрасту/роду занятий/месту проживания и т.д.
			\item[] \(p_3 z_2 x_2\) - анализ имеющегося опыта на предмет наличия полезных для текущего проекта навыках.
			\end{itemize}
