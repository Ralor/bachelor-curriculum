	\chapter{Экспертная оценка свойств системы}
	\section{Описание}
		\paragraph{Цель работы}
			Получить практические навыки экспертного оценивания 
			систем различными методами и обработки результатов оценивания.
		\paragraph{Формируемые компетенции}
			\begin{itemize}
			\item владение культурой мышления, способность к обобщению, 
			анализу, восприятию информации, постановке цели и выбору путей её достижения (ОК-1) 
			\item способность находить организационно-управленческие 
			решения в нестандартных ситуациях и готовность нести за них ответственность (ОК-4)
			\end{itemize}		
		\paragraph{Самостоятельная работа}
			Изучение методов выявления предпочтений экспертов (ранжирования, 
			парных сравнений, непосредственной оценки, последовательного сравнения).

			Оценим согласованность экспертов через коэффициент конкордации 
			
			\[
			K = \frac{
				12 \sum\limits_{j=1}^n \sum\limits_{i=1}^m 
					( \overline{r} - r_{ij} )^2
			}{
				m^2(n^3-n) - m\sum\limits_{i=1}^{m}T_i
			} = 
			\frac{
				12 \sum\limits_{j=1}^4 \sum\limits_{i=1}^3 
					( 7.5 - r_{ij} )^2
			}{
				3^2(4^3-4) - 3*36
			} = 0.48
			\]

			где 
			
			\begin{enumerate}	\itemsep 0pt \parskip 0pt \parsep 0pt 
			\item[] \(m\) - количество экспертов
			\item[] \(n\) - количество объектов
			\item[] \(\overline{r}\) - средний ранг
			\item[] \(T_i\) - показатель связанных рангов в \(i\)-й ранжировке,\\
				\(T_i~=~\sum\limits_{k=1}^{H_i}{h_k}^3 - h_k\), 
					где 
					\begin{enumerate} \itemsep 0pt \parskip 0pt \parsep 0pt										
					\item[] \(H_i\) - число групп равных рангов в \(i\) - й группе
					\item[] \(h_k\) - число равных рангов в \(k\) - й группе связанных рангов
					\end{enumerate}
			\end{enumerate}		
				
			На основе вычисленного \(K\) дадим качественную характеристику согласованности мнений экспертов ***ОШИБКА СМ. TEX***

		\paragraph{Парные сравнения систем}	
			Каждый эксперт составляет матрицу парных сравнений, элемент которой равен
			\begin{multicols}{2}
			\[w_{ij} = 
				\begin{cases}
				1, & x_i \succeq x_j\\
				0, & x_i \prec x_j
				\end{cases}
			\]
			\vfill\columnbreak
			Условия согласованности:
			\begin{enumerate} \itemsep 0pt \parskip 0pt \parsep 0pt
				\item[] \(w_{ii} = 1\)
				\item[] \(w_{ij} = 1 \Rightarrow w_{ji} = 0\)
				\item[] \(w_{ij} = 1 \land w_{jk} = 1 \Rightarrow w_{ik} = 1\)
			\end{enumerate}
			\end{multicols}
